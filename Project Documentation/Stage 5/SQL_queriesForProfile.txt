 SELECT AVG(A_MEAN), OCC_TITLE FROM OCC_STATS
 WHERE OCC_TITLE  LIKE  '%Computer and Information Systems Managers';
 
 SELECT STATE_ID, TOT_EMP FROM OCC_STATS
 WHERE OCC_TITLE LIKE  '%Computer and Information Systems Managers'
 GROUP BY TOT_EMP DESC, STATE_ID
 LIMIT 10;
 
 SELECT S1.OCC_TITLE, S3.LEVELDESCRIPTION FROM OCC_STATS S1, OCCUPATION S2, EDUCATIONALLEVEL S3
 WHERE S1.OCC_CODE = S2.OCC_CODE AND S2.OCC_TITLE  LIKE '%Computer and Information Systems Managers'  AND S3.EDUCATIONLEVELID = S2.EDUCATIONLEVELID;
 
 SELECT OCC_TITLE, EDUCATIONLEVELID FROM OCCUPATION
 WHERE OCC_TITLE  LIKE '%Computer and Information Systems Managers';
